# Sol.1   

語言: C++  
解題思路：雙迴圈暴力解  
時間複雜度：O(n^2)    
空間複雜度：O(1)

```cpp
class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        int size = nums.size();

        // 第一個遍歷迴圈
        for (int i = 0; i < size; i++) {
            //發現移除目標
            if(nums[i] == val) {
                //將後數組全部前移 
                for (int j = i + 1; j < size; j++) {
                    nums[j-1] = nums[j];
                }

                size--; //總長減少1
                i--;  //全部回退，第i個也會受到回退影響
            }
        }

        return size;
    }
};
```

# Sol.2

語言: C++  
解題思路：快慢指針  
時間複雜度：O(n)  
空間複雜度：O(1)

```cpp
class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        int slow = 0;
        int fast = 0;
        int size = nums.size();

        while (fast != nums.size()) {
            if (nums[fast] != val) {
                nums[slow] = nums[fast];
                slow++;
                fast++;
                continue;
            } else {
                fast++;
                size--;
            }
        }
        return size;
    }
};
```
