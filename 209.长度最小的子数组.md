# Sol.1   

語言: C++  
解題思路：雙For迴圈暴力解
時間複雜度：O(n^2)    
空間複雜度：O(1)

```c++
class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums) {
        
        int result = INT32_MAX;
        int size = nums.size();
        int maxSize = nums.size();

        for (int i = 0; i < size; i++) {
            int sum = 0;
            for (int j = i; j < size; j++) {
                sum += nums[j];
                if (sum >= target) {
                    int length = j - i + 1;
                    result = result < length ? result : length;
                    break;
                }
            }
        }

        return result == INT32_MAX ? 0 : result ;
    }
};
```
