59. 螺旋矩阵 II

[KamaCoder網站](https://leetcode.cn/problems/spiral-matrix-ii/description/)

# Sol.1   

語言: C++  
解題思路：前綴合  
時間複雜度：O(n^2)    
空間複雜度：O(n^2)  

```cpp
class Solution {
public:
    vector<vector<int>> generateMatrix(int n) {
        
        vector<vector<int>> matrix(n,vector<int>(n,0));
        int mid = n/2; //中間單獨格的位置，只有n是奇數有中間單獨格
        int loop = n/2; //循環圈數
        int count = 1;
        int offset = 1;
        int i,j;
        int start_x = 0;
        int start_y = 0;

        while(loop--) {
            i = start_x;
            j = start_y;

            for(; j < n-offset; j++) {
                matrix[i][j] = count++;
            }

            for(; i < n-offset; i++) {
                matrix[i][j] = count++;
            }

            for(; j > start_y; j--) {
                matrix[i][j] = count++;
            }

            for(; i > start_x; i--) {
                matrix[i][j] = count++;
            }

            start_x++;
            start_y++;
            offset++;
        }

        if (n % 2) {
            matrix[mid][mid] = count;
        }

        return matrix;
    }
};
```
